<script>

(function () {
  
  // CSS selectors for icons, boxes, and grid container
  const ICON_SELECTOR = '.expand-icon';
  const BOX_SELECTOR = '.expandable-box';
  const GRID_SELECTOR = '.expandable-section';

  /*Function to find how many columns the grid has.*/
  function getColumnsCount(grid) {
    const style = window.getComputedStyle(grid);
    const cols = style.gridTemplateColumns;
    return cols ? cols.split(/\s+/).filter(Boolean).length : 4;
  }

  /*Move a box to the start of its row in the grid*/
  function moveBoxToRowStart(box) {
    const grid = box.closest(GRID_SELECTOR);
    if (!grid) return;

    // Collect all boxes in the grid here that match the expandable box
    const boxes = Array.from(grid.children).filter(n => n.matches(BOX_SELECTOR));

    const rowsMap = new Map(); //rowsmap groups boxes by row
    const TOL = 6; // px tolerance for detecting same row, allow small diffs

    boxes.forEach(b => {
      const top = Math.round(b.getBoundingClientRect().top);
      let foundKey = null;

      // Check if this box belongs to an existing row
      for (let key of rowsMap.keys()) {
        if (Math.abs(key - top) <= TOL) { foundKey = key; break; }
      }

      const key = (foundKey !== null) ? foundKey : top;
      if (!rowsMap.has(key)) rowsMap.set(key, []);
      rowsMap.get(key).push(b);
    });

    // Find the row of the current box
    const boxTop = Math.round(box.getBoundingClientRect().top);
    let boxRowKey = null;
    for (let key of rowsMap.keys()) {
      if (Math.abs(key - boxTop) <= TOL) { boxRowKey = key; break; }
    }
    if (boxRowKey === null) return;

    const rowBoxes = rowsMap.get(boxRowKey) || [];
    if (!rowBoxes.length) return;

    // Move box to the start of its row
    const firstInRow = rowBoxes[0];
    if (firstInRow !== box) {
      grid.insertBefore(box, firstInRow);
    }
  }

  /*Expand a box to full width and show its sub-boxes*/
  function expandBox(box, icon) {
  moveBoxToRowStart(box);
  box.classList.remove('collapsed');
  box.classList.add('expanded');
  box.style.gridColumn = '1 / -1'; // span all columns
  if (icon) icon.textContent = 'remove';

  const subBoxes = box.querySelector('.sub-boxes');
  if (subBoxes) {
    subBoxes.classList.remove('hidden');
    subBoxes.style.gridTemplateColumns = 'repeat(5, 1fr)';
    //adjust grid columns based on child count
    // const subBoxCount = subBoxes.children.length;
    // if (subBoxCount > 4) {
    //   subBoxes.style.gridTemplateColumns = 'repeat(3, 1fr)';
    // } else {
    //   subBoxes.style.gridTemplateColumns = 'repeat(4, 1fr)';
    // }
  }
}
  //Collapse a box back to normal size and hide its sub-boxes
  function collapseBox(box, icon) {
    box.classList.add('collapsed');

    // After animation, reset styles
    setTimeout(() => {
      box.classList.remove('expanded', 'collapsed');
      box.style.gridColumn = '';
    }, 400);

    if (icon) icon.textContent = 'add';

    const subBoxes = box.querySelector('.sub-boxes');
    if (subBoxes) subBoxes.classList.add('hidden');
  }

  // Attach click handlers to all expand icons
  document.querySelectorAll(ICON_SELECTOR).forEach(icon => {
    icon.addEventListener('click', () => {
      const box = icon.closest(BOX_SELECTOR);
      if (!box) return;

      if (box.classList.contains('expanded')) {
        collapseBox(box, icon);
      } else {
        expandBox(box, icon);
      }
    });
  });

  // Handle window resize: re-align expanded boxes
  let resizeTimer = null;
  window.addEventListener('resize', () => {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(() => {
      document.querySelectorAll(`${GRID_SELECTOR} ${BOX_SELECTOR}.expanded`).forEach(box => {
        moveBoxToRowStart(box);
        box.style.gridColumn = '1 / -1';
      });
    }, 120);
  });
})();

//Equalize row heights in the grid
function equalizeRowHeights() {
  const grid = document.querySelector('.expandable-section');
  if (!grid) return;

  const boxes = Array.from(grid.children).filter(n => n.matches('.expandable-box'));
  const rowsMap = new Map();
  const TOL = 6;

  // Group boxes by row
  boxes.forEach(b => {
    b.style.minHeight = 'auto'; // reset height
    const top = Math.round(b.getBoundingClientRect().top);
    let foundKey = null;

    for (let key of rowsMap.keys()) {
      if (Math.abs(key - top) <= TOL) { foundKey = key; break; }
    }

    const key = (foundKey !== null) ? foundKey : top;
    if (!rowsMap.has(key)) rowsMap.set(key, []);
    rowsMap.get(key).push(b);
  });

  // Set equal height for each row
  rowsMap.forEach(rowBoxes => {
    const maxHeight = Math.max(...rowBoxes.map(b => b.offsetHeight));
    rowBoxes.forEach(b => b.style.minHeight = `${maxHeight}px`);
  });
}
