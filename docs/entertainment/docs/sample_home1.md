<script>
  function setUniformHeight() {
    const containers = document.querySelectorAll('.section-containers .custom-container');
    let maxHeight = 0;

    // Reset heights first
    containers.forEach(container => container.style.height = 'auto');

    // Find max height
    containers.forEach(container => {
      const height = container.offsetHeight;
      if (height > maxHeight) maxHeight = height;
    });

    // Apply max height to all
    containers.forEach(container => container.style.height = maxHeight + 'px');
  }

  // Run on load and on resize
  window.addEventListener('load', setUniformHeight);
  window.addEventListener('resize', setUniformHeight);
</script>
